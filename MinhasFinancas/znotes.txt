    public static void removerUser(AppDbContext context)
    {
        var user = new User
        {
            Id = 5,
            Nome = "Eduardo",
            Email = "1eduardofadel@gmail.com",
            Senha = "senha123"
        };
        context.Users.Remove(user);
        context.SaveChanges();
    }

    public static void imprimirUsers(AppDbContext context)
    {
        var userList = context.Users.ToList();
        foreach (var u in userList)
        {
            Console.WriteLine("[" + u.Id + "] " + u.Nome + "\t" + u.Email);
        }
    }

    public static User getById(AppDbContext context, int id)
    {
        var user = context.Users.Find(id);
        if (user == null)
        {
            Console.WriteLine("Nenhum usuário com este id!");
        }

        return user;
    }
